--create person table

CREATE TABLE PERSON (
Personal_code VARCHAR PRIMARY KEY,
Name VARCHAR NOT NULL,
LastName VARCHAR NOT NULL 
);

--create bank table

CREATE TABLE BANK (
Id INT PRIMARY KEY,
Name VARCHAR NOT NULL
);

-- create account table

CREATE TABLE ACCOUNT(
Id INT PRIMARY KEY,
Personal_code VARCHAR REFERENCES PERSON(Personal_code),
Bank_id INT REFERENCES BANK(Id)
);

--create card table

CREATE TABLE CARD (
Id INT PRIMARY KEY,
Account_id INT REFERENCES ACCOUNT(ID),
Type VARCHAR CHECK (Type ='Credit' OR Type ='Debit'),
Balance DECIMAL(7,2) NOT NULL,
Debt DECIMAL(7,2) Default 0.00
);
-- create transaction table

CREATE TABLE TRANSACTION (
Id INT PRIMARY KEY,
Card_id INT REFERENCES CARD(ID),
Date DATE NOT NULL,
Amount DECIMAL(7,2) NOT NULL
);
